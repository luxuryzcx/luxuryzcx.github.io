    for i in range(12):
        # 计算当前要使用的固定字符串
        current_str = fixed_strings[i % 3]
        
        # 计算当前要比较的字符位置
        char_pos = 2 * (i // 3)
        
        # 获取固定字符串中的字符
        fixed_char = current_str[char_pos]
        
        # 根据条件计算输入字符串中对应位置的字符
        # 因为条件是：fixed_char - input_char == 1
        # 所以 input_char = fixed_char - 1
        input_char = chr(ord(fixed_char) - 1)
        
        # 将计算得到的字符添加到结果字符串
        result_string += input_char
    
    return result_string
